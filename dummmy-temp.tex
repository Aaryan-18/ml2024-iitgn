\begin{columns}
	\pause \begin{column}{0.6\textwidth}
		\begin{adjustbox}{max totalsize={\textwidth},center}
			
			\begin{tikzpicture}
			\begin{axis}[colorbar,xlabel=$x$, ylabel=$y$, zlabel=$z$,title={Surface Plot}]
			\addplot3[
			surf,
			]
			{x^2+y^2};
			\end{axis}
			\end{tikzpicture}
		\end{adjustbox}
		
	\end{column}
	\pause \begin{column}{0.5\textwidth}
		\begin{adjustbox}{max totalsize={\textwidth},center}
			\begin{tikzpicture}
			\begin{axis}
			[
			title={Contour plot, view from top},
			view={0}{90},
			xlabel=$x$,
			ylabel=$y$,
			axis x line*=bottom,
			axis y line*=left,
			xtick align=outside,
			ytick align=outside,
			unit vector ratio*=1 1 1,
			]
			\addplot3[
			contour gnuplot={number=20,}
			]
			{x^2+y^2};
			\addplot3[black,
			quiver={
				u={2*x},
				v={2*y},
				scale arrows=0.1,
			},
			-stealth,samples=10]
			{x^2+y^2};
			\end{axis}
			\end{tikzpicture}
		\end{adjustbox}
	\end{column}
\end{columns}
